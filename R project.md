#first we need to cut the time into hours as its our main interest.
hour <- function(data) {
  start_time <-
    sapply(strsplit(as.character(data$Start.Time), " "), "[", 2)
  #suset hour from the data
  hour <- substr(x = start_time, 1, 2)
}
